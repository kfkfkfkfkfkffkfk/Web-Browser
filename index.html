function navigate() { if (!activeTabId) return; let url = document.getElementById('url-input').value.trim(); if (!url) return; // Basic check to add https:// if the user forgot it if (!url.startsWith('http://') && !url.startsWith('https://')) { url = 'https://' + url; } // THE FIX: Wrap the URL in a free public proxy to bypass X-Frame-Options // encodeURIComponent ensures special characters in the URL don't break the proxy const proxiedUrl = 'https://corsproxy.io/?' + encodeURIComponent(url); // Update the iframe with the proxied URL const activeIframe = document.getElementById('view-' + activeTabId); activeIframe.src = proxiedUrl; // Update tab title const activeTabBtn = document.getElementById('btn-' + activeTabId); const tabTitle = activeTabBtn.querySelector('.tab-title'); // Extract a simple title from the original URL (e.g., "example.com") try { const domain = new URL(url).hostname; tabTitle.innerText = domain; } catch (e) { tabTitle.innerText = url; } }
